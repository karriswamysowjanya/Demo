from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.common.keys import Keys
import time

@given("user opens Flipkart homepage")
def step_open_flipkart(context):
    context.driver = webdriver.Chrome()
    context.driver.maximize_window()
    context.driver.get("https://www.flipkart.com")
    time.sleep(3)

    # Close login popup if present
    try:
        context.driver.find_element(By.XPATH, "//button[text()='âœ•']").click()
    except:
        pass

@when('user searches for "iPhone"')
def step_search_product(context):
    search_box = context.driver.find_element(By.NAME, "q")
    search_box.send_keys("iPhone")
    search_box.send_keys(Keys.ENTER)
    time.sleep(3)

@then("search results should be displayed")
def step_validate_results(context):
    results = context.driver.find_elements(By.XPATH, "//div[contains(text(),'iPhone')]")
    assert len(results) > 0, "Search results not displayed"
    context.driver.quit()
